57 void DailyQA::thresholds()
58 {
59  std::string cached_company_name;
60  for(auto&& line : threshold_sheet)
61  {
62  auto company = line[0], project = line[1],
63  _1threshold = line[2], _2threshold = line[3];
64 
65  //Caches company name for projects w/ multiple carriers, see morning_QA()
66  if(company != "") cached_company_name = company;
67  else if(company == "" && project != "") company = line[0] = cached_company_name;
68  else if(company == "" && project == "_") { continue; }
69 
70  //Remove '%' character
71  _1threshold.pop_back(); _2threshold.pop_back();
72 
73  //Lookup based on concatenation of company and project
74  auto str = company+project;
75  if(company != "_" && project != "_")
76  threshold_entries.insert({str, {std::stod(_1threshold), std::stod(_2threshold)}});
77  }

89 void DailyQA::add_name_entries()
90 {
91  std::string cached_company_name;
92  for(auto&& line : names_sheet)
93  {
94  auto company = line[0], project = line[1];
95 
96  // The projects don't all have a company name listed next to them
97  // For example, if the company has multiple carriers it only lists the
98  // name next to the first project. This caches that name so that all
99  // projects under the same company have a company name listed when
100  // the data is entered into the output file.
101  if(company != "") cached_company_name = company;
102  else if(company == "" && project != "") company = line[0] = cached_company_name;
103  else if(company == "" && project == "_") { continue; }
104 
105  // The lookup uses the concatenation of the company and the project names
106  // Then, insert into name_entries hash table for later usage in setting up throughput sheet.
107  auto str = company+project;
108  if(company != "_" && project != "_") name_entries.insert({str, line});
109  }

112 void DailyQA::morning_QA()
113 {

118  std::string cached_company_name;
119  for(auto&& line : names_sheet)
120  {
121  auto company = line[0], project = line[1];
122  auto str = company+project;
123 
124  static int ws_count = 0; //Whitespace count (merely for logging purposes)
125  if(auto search = data_entries.find(str); search != std::end(data_entries) /*&& search->first != ""*/)
126  {
127  #define out(x,y) ( (x)[y] == ""? "n/a" : (x)[y] )
128  auto const& _7 = out(search->second,4);
129  auto const& _3 = out(search->second,6);
130  auto const& _2 = out(search->second,8);
131  auto const& _1 = out(search->second,10);
132 
133  auto t_1 = find_threshold(_1);
134  auto t_2 = find_threshold(_2);
135 
136  outfile << out(search->second,0) << ',' << out(search->second,1) << ','
137  << _7 << ',' << _3 << ','<< _2 << ',' << _1;
138  if(auto s = threshold_entries.find(str); s != std::end(threshold_entries))
139  {
140  auto const& match = s->second;
141  bool cond = (match.first > t_1 && match.second > t_2);
142  if(throughput_entries.find(str) != std::end(throughput_entries))
143  outfile << ",***,";
144  else
145  outfile << ",n/a,";
146  outfile << (cond?"***,,***,\n":"no,n/a,no,n/a\n");
147  }
148  else
149  {

151  << " does not have a corresponding threshold entry\n";
152  outfile << ",,ERR,ERR,ERR,ERR\n";
153  }
154  #undef out
155  }
156  else if(project == "_" || company == "_" || project == "" || company == "")

169 void DailyQA::throughput()
170 {
171  //Counters
172  int total_ATT{}, total_verizon{}, total_sprint{}, total_tmobile{}, total_entries{};
173  int only_ATT{}, only_verizon{}, only_sprint{}, only_tmobile{};
174  int all{};
175 
176  std::vector<Line> D2S{};
177 
178  std::regex r{R"~(\‍(([0-9]+)\).+)~"}; //Regular expression to extract provider number
179  int unknown_providers{};
180 
181  //Column titles
182  t_outfile << ",Company,Project,Campaign,Drip Name,Drip ID, ATT, Sprint,T-Mobile,Verizon,Tested,Edited\n";
183 
184  //Delete unnecessary projects on throughput sheet
185  for(auto&& line : throughput_sheet)
186  {
187  bool ATT{}, verizon{}, sprint{}, tmobile{};
188  auto str = line[1] + line[2]; //Lookup based on concatenation, as before
189  if(auto search = name_entries.find(str); search != std::end(name_entries))
190  {
191  ++total_entries;
192 
193  //Add to list of projects that will be in final throughput sheet
194  D2S.push_back(line);
195 
196  #define provider_block()\
197  {\
198  if(std::smatch m; std::regex_match(dat, m, r))\
199  {\
200  if(auto search = provider_entries.find(m[1]); search != std::end(provider_entries)) ++search->second.second;\
201  else { ++unknown_providers; provider_entries.insert({m[1],{"UNKNOWN",1}}); }\
202  }\
203  }
204 
205  //Update total blocks and provider blocks
206  if(auto dat = line[6]; dat != "" && dat != "_"){ ATT = ++total_ATT; provider_block(); }
207  if(auto dat = line[7]; dat != "" && dat != "_"){ sprint = ++total_sprint; provider_block(); }
208  if(auto dat = line[8]; dat != "" && dat != "_"){ tmobile = ++total_tmobile; provider_block(); }
209  if(auto dat = line[9]; dat != "" && dat != "_"){ verizon = ++total_verizon; provider_block(); }
210 
211  #undef provider_block
212 
213  //Update exclusive blocks
214  if(ATT && !sprint && !tmobile && !verizon) ++only_ATT;
215  if(!ATT && sprint && !tmobile && !verizon) ++only_sprint;
216  if(!ATT && !sprint && tmobile && !verizon) ++only_tmobile;
217  if(!ATT && !sprint && !tmobile && verizon) ++only_verizon;
218 
219  //Update blocks on all carriers
220  if(ATT && sprint && tmobile && verizon) ++all;
221  }
222  }
223 
224  //Lines in throughput sheet are lexicographically sorted.
225  //Stable sort is used to preserve relative order of elements, but is slightly less efficient
226  std:: sort(std::begin(D2S), std::end(D2S), [](auto&& a, auto&& b){ return a[3] < b[3]; }); //First sort by campaign name
227  std::stable_sort(std::begin(D2S), std::end(D2S), [](auto&& a, auto&& b){ return a[2] < b[2]; }); //Then sort by project name
228  std::stable_sort(std::begin(D2S), std::end(D2S), [](auto&& a, auto&& b){ return a[1] < b[1]; }); //Finally sort by company name
229 
230  //Output only D2S projects that are on the names sheet.
231  for(auto&& line : D2S)
232  for(int i = 0; i < 10; ++i) { auto val = line[i]; t_outfile << (val=="_"?" ":val) << (i < 9?',':'\n'); }
233 
234  //Output stats (to file and stdout)
235  std::cout << "\u001b[32;1mThroughput sheet successfully formatted.\u001b[0m\n Outputting statistics to stdout...\n\n";
236 
237  auto tp_sz = throughput_sheet.size();
238  std::ostream* out = &std::cout;
239  bool outflag = false;
240  if(!log) std::cerr << "\u001b[31;1mError: could not open \u001b[35;1moutput/log.csv\u001b[0m\n";
241 DUMP:
242  (*out) << (!outflag?"\u001b[37;1m":"") << "Number of entries before formatting: " << (!outflag?"\u001b[0m":",") << tp_sz << "\n";
243  (*out) << (!outflag?"\u001b[37;1m":"") << "Total number of entries deleted: " << (!outflag?"\u001b[0m":",") << tp_sz - total_entries << "\n";
244  (*out) << (!outflag?"\u001b[37;1m":"") << "Final size of throughput sheet (entries): " << (!outflag?"\u001b[0m":",") << total_entries << "\n\n";
245 
246  (*out) << (!outflag?"\u001b[37;1m":"") << "Total ATT Blocks: " << (!outflag?"\u001b[0m":",") << total_ATT << '\n';
247  (*out) << (!outflag?"\u001b[37;1m":"") << "Total Verizon Blocks: " << (!outflag?"\u001b[0m":",") << total_verizon << '\n';
248  (*out) << (!outflag?"\u001b[37;1m":"") << "Total Sprint Blocks: " << (!outflag?"\u001b[0m":",") << total_sprint << '\n';
249  (*out) << (!outflag?"\u001b[37;1m":"") << "Total T-Mobile Blocks: " << (!outflag?"\u001b[0m":",") << total_tmobile << "\n\n";
250 
251  (*out) << (!outflag?"\u001b[37;1m":"") << "Exclusive ATT Blocks: " << (!outflag?"\u001b[0m":",") << only_ATT << '\n';
252  (*out) << (!outflag?"\u001b[37;1m":"") << "Exclusive Verizon Blocks: " << (!outflag?"\u001b[0m":",") << only_verizon << '\n';
253  (*out) << (!outflag?"\u001b[37;1m":"") << "Exclusive Sprint Blocks: " << (!outflag?"\u001b[0m":",") << only_sprint << '\n';
254  (*out) << (!outflag?"\u001b[37;1m":"") << "Exclusive T-Mobile Blocks: " << (!outflag?"\u001b[0m":",") << only_tmobile << "\n\n";
255 
256  (*out) << (!outflag?"\u001b[37;1m":"") << "All carrier Blocks: " << (!outflag?"\u001b[0m":",") << all << "\n\n";
257 
258  for(auto&& [code, provider] : provider_entries)
259  (*out) << (!outflag?"\u001b[37;1m":"") << "Total " << provider.first << " (" << code << ") blocks:" << (!outflag?"\u001b[0m ":",") << provider.second << '\n';
260  if(!outflag && log)
261  {
262  outflag = true;
263  out = &log;
264  std::cout << "\n\u001b[32;1mDone.\n\u001b[0m\nOutputting stats to \u001b[35;1moutput/log.csv\u001b[0m...";
265  goto DUMP;
266  }
267 
268  if(unknown_providers > 0) std::cerr << "\n\u001b[31;1m" << "There are " << unknown_providers << " unknown providers. Please update \u001b[35;1minput/providers.csv\u001b[0m" << '\n';
269  std::cout.put('\n');
270 }
